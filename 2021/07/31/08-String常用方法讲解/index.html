String常用方法

equals 和 ==

public static void main(String[] args){
    /*
    	equals:只能使用与引用类型的"比较",按照指定规则进行比较
    		   如果根据指定规则比较的结果一样则返回true,否则返回false
    	  ==  :用于基本类型的比较,直接比较值
    	    用于引用类型的比较,比较的是地址
    */
    /*
    	String是一个不可变的字符序列,如果要使String对象的字面值发生改变,则相当于创建了一个新的String对象
    	Java语言为String类型的对象提供了一个常量池
    */
    String s1 = "hello"; //hello是字面值
    String s2 = "hello";
    String s3 = new String("hello"); //开辟了一个新的对象,不在常量池里
    System.out.println("s1.equals(s2)="+s1.equals(s2)); //true
    System.out.println("s1.equals(s3)="+s1.equals(s3)); //true
    System.out.println("s1==s2 =>"+(s1==s2)); //true
    System.out.println("s1==s3 =>"+(s1==s3)); //false
    
}

一共有几个对象?

String s1 = "hello"; //1个

String s3 = "hello"+"world"; //world一个,helloworld一个
String s4 = new String("helloworld"); //一个,一共四个

几个常用String方法

public class TestString{

    public static void main(String[] args) {
        
        int age = 25;
        String content = "Java is very hard!";
        
		// char charAt(int index):返回指定索引处的 char 值。
        char c = content.charAt(5);
		System.out.println("content.charAt(5) =>"+c);
		
		//int compareTo(String anotherString):按字典顺序比较两个字符串。
		System.out.println("content.compareTo(\"Java is very hard!\") =>"+content.compareTo("Java is very hard!"));
		
		//int compareToIgnoreCase(String str):按字典顺序比较两个字符串，不考虑大小写。 
		System.out.println("content.compareToIgnoreCase(\"java is very hard!\") =>"+content.compareToIgnoreCase("java is very hard!"));
		
		//boolean contains(CharSequence s):当且仅当此字符串包含指定的 char 值序列时，返回 true。 
		System.out.println("content.contains(\"very\") =>"+content.contains("very"));
		
		//boolean endsWith(String suffix):测试此字符串是否以指定的后缀结束。 
		System.out.println("content.endsWith(\"!\") =>"+content.endsWith("!"));
		
		//boolean equals(Object anObject):将此字符串与指定的对象比较。 
		System.out.println("content.equals(\"Java is very hard!\") =>"+content.equals("Java is very hard!"));
		
		//boolean equalsIgnoreCase(String anotherString):将此 String 与另一个 String 比较，不考虑大小写。 
		System.out.println("content.equalsIgnoreCase(\"Java is very hard!\") =>"+content.equalsIgnoreCase("Java is very hard!"));
		
		//byte[] getBytes():使用平台的默认字符集将此 String 编码为 byte 序列，并将结果存储到一个新的 byte 数组中。 
		System.out.print("content.getBytes()==>");
		byte[] bytes = content.getBytes();
		for(int i=0;i<bytes.length;i++){
			System.out.print(bytes[i]);
			if(i<bytes.length-1){
				System.out.print(",");
			}else{
				System.out.print("\r\n");
			}
		}
		
		//int hashCode():返回此字符串的哈希码
		System.out.println("content.hashCode()==>"+content.hashCode());
		//得到的是content引用变量指向的String对象的内存空间所表示的地址的数字表现形式经过哈希算法得到的数值
		
		//int indexOf(int ch):返回指定字符在此字符串中第一次出现处的索引。 
		System.out.println("content.indexOf('a') =>"+content.indexOf('a'));
		System.out.println("content.indexOf(97) =>"+content.indexOf(97));
		
		//int indexOf(String str):返回指定子字符串在此字符串中第一次出现处的索引。 
		System.out.println("content.indexOf(\"is\") =>"+content.indexOf("is"));
		
		//boolean isEmpty():当且仅当 length() 为 0 时返回 true。 
		System.out.println("content.isEmpty()==>"+content.isEmpty());
		
		//int lastIndexOf(int ch):返回指定字符在此字符串中最后一次出现处的索引。 
		System.out.println("content.lastIndexOf('a')==>"+content.lastIndexOf('a'));
		
		//int lastIndexOf(String str):返回指定子字符串在此字符串中最右边出现处的索引。 
		System.out.println("content.lastIndexOf(\"is\")==>"+content.lastIndexOf("is"));

		//int length() :返回此字符串的长度。
		System.out.println("content.length() =>"+content.length());
		
		//String replace(char oldChar, char newChar):返回一个新的字符串，它是通过用 newChar 替换此字符串中出现的所有 oldChar 得到的。
		System.out.println("content.replace('a','A')==>"+content.replace('a','A'));
		
		//String replace(CharSequence target, CharSequence replacement) :使用指定的字面值替换序列替换此字符串所有匹配字面值目标序列的子字符串。 
		System.out.println("content.replace(\"Java\",\"<-_->\")==>"+content.replace("Java","<-_->"));
		
		//String replaceAll(String regex, String replacement):使用给定的 replacement 替换此字符串所有匹配给定的正则表达式的子字符串。 
		String str1 = "1a2b3cd.23e";
		System.out.println("str1.replace(\".\",\"。\") ==>"+str1.replace(".","。"));
		System.out.println("str1.replaceAll(\".\",\"。\") ==>"+str1.replaceAll(".","。")); //正则表达式.匹配除换行符以外的任意字符
		System.out.println("str1.replaceAll(\"\\d\",\"锐\") ==>"+str1.replaceAll("\\d","锐")); //	\d匹配数字
		System.out.println("str1.replace(\"e\",\"哈\") ==>"+str1.replace("e","哈")); //不是regex,所以精确匹配该字符
		
		//String replaceFirst(String regex, String replacement):使用给定的 replacement 替换此字符串匹配给定的正则表达式的第一个子字符串。
		System.out.println("content.replaceFirst(\"a\",\"哈\") ==>"+content.replaceFirst("a","哈"));
		
		//String[] split(String regex):根据给定正则表达式的匹配拆分此字符串。 
		String[] stringArray = content.split(" ");//空格作为分割符
		System.out.print("content.split(\" \")==>");
		for(int i=0;i<stringArray.length;i++){
			System.out.print(stringArray[i]);
			if(i<stringArray.length-1){
				System.out.print(",");
			}else{
				System.out.print("\r\n");
			}
				
		}
		
		//boolean startsWith(String prefix):测试此字符串是否以指定的前缀开始。
		System.out.println("content.startsWith(\"Java\")==>"+content.startsWith("Java"));
		
		//String substring(int beginIndex):返回一个新的字符串，它是此字符串的一个子字符串。 
		System.out.println("content.substring(5)==>"+content.substring(5));
		
		
		//String substring(int beginIndex, int endIndex):返回一个新字符串，它是此字符串的一个子字符串。 
		System.out.println("content.substring(5,10)==>"+content.substring(5,10));//左包含,右不包含
		System.out.println("content.substring(5,content.length())==>"+content.substring(5,content.length()));
		
		//char[] toCharArray():将此字符串转换为一个新的字符数组 
		char[] charArray = content.toCharArray();
		System.out.print("content.toCharArray()==>");
		for(int i=0;i<charArray.length;i++){
			System.out.print(charArray[i]);
			if(i<charArray.length-1){
				System.out.print(",");
			}else{
				System.out.print("\r\n");
			}
				
		}
		
		//String toLowerCase():使用默认语言环境的规则将此 String 中的所有字符都转换为小写。 
		System.out.println("content.toLowerCase()==>"+content.toLowerCase());
		
		//String toUpperCase():使用默认语言环境的规则将此 String 中的所有字符都转换为大写。 
		System.out.println("content.toUpperCase()==>"+content.toUpperCase());
		
		//String toString():返回此对象本身(它已经是一个字符串！) 
		System.out.println("content.toString()==>"+content.toString());
		System.out.println("content==>"+content);//和上面一样
		
		//String trim():返回字符串的副本，忽略前导空白和尾部空白 
		System.out.println("\"  老子 不想学 java  \".trim()==>"+"  老子 不想学 java  ".trim());
		
		//static String valueOf(xxx b):将xxx类型的b值转成String类型
		int a1 = 100;
		double a2 = 3.14;
		boolean a3 = true;
		char a4 = 'A';
		
		String s1 = String.valueOf(a1);  //"100"  静态方法,不用对象调用,用类名调用
		String s2 = String.valueOf(a2);  //"3.14"
		String s3 = String.valueOf(a3);  //"true"
		String s4 = String.valueOf(a4);  //"A"
		
		String s5 = ""+a1;  //推荐这种方式
		String s6 = ""+a2;
		String s7 = ""+a3;
		String s8 = ""+a4;		
		
		System.out.println("s1="+s1);
		System.out.println("s2="+s2);
		System.out.println("s3="+s3);
		System.out.println("s4="+s4);
		
		System.out.println("s5="+s5);
		System.out.println("s6="+s6);
		System.out.println("s7="+s7);
		System.out.println("s8="+s8);
    }

}
