1.是否闰年

/*
1、编写程序，判断给定的某个年份是否是闰年，给定的
年份通过用户输入的方式动态给定，输入的方式如：
Scanner scan = new Scanner(System.in);
int year = scan.nextInt();
闰年的判断规则如下：
 若某个年份能被4整除但不能被100整除，则是闰年
 若某个年份能被400整除，则也是闰年
*/
import java.util.Scanner;

public class Homework1{
	
	public static void main(String[] args){
		
		Scanner scanner = new Scanner(System.in);
		while(true){
			System.out.println("请输入一个有效年份");
			int input = scanner.nextInt();
			if(input<0){
				System.out.println("年份错误");
				System.exit(0);	
			}
			if((input%4==0&input%100!=0) || input%400==0){
				System.out.println(input+"是闰年");
			}else{
				System.out.println(input+"是平年");
			}
		}
	}
	
}

3.利用三种循环循环打印 9*9 乘法表

/*
利用三种循环循环打印 9*9 乘法表

1*1=1       		相当于j*i=
1*2=2 2*2=4 
1*3=3 2*3=6 3*3=9 
1*4=4 2*4=8 3*4=12 4*4=16
*/
public class Homework3{
	
	public static void main(String[] args){
		
		System.out.println("=== for 9*9 ===");
		for(int i=1;i<=9;i++){ //列
			for(int j=1;j<=i;j++){ //行
				System.out.print(j+"*"+i+"="+j*i);
				if(j==i){
				System.out.print("\r\n");
				}else{
					System.out.print("\t");
				}
			}
		}
		
		System.out.println("=== while 9*9 ===");
		int i=1;
		while(i<=9){
			int j=1;
			while(j<=i){
				System.out.print(j+"*"+i+"="+j*i);
				if(j==i){
					System.out.print("\r\n");
				}else{
					System.out.print("\t");
				}
				j++;
			}
			i++;
		}
		
		System.out.println("=== do while 9*9 ===");
		i=1;
		do{
			int j=1;
			do{
				System.out.print(j+"*"+i+"="+j*i);
				if(j==i){
					System.out.print("\r\n");
				}else{
					System.out.print("\t");
				}
				j++;
			}while(j<=i);
			i++;
		}while(i<=9);
		
		
	}
}

4.给定一个百分制的分数，输出相应的等级

/*
给定一个百分制的分数，输出相应的等级
90分以上 A级
80~89 B级
70~79 C级
60~69 D级
60分以下 E级
*/
public class Homework4{
	
	public static void main(String[] args){
		
		//String -> int
		int score = Integer.parseInt(args[0]);
		switch(score/10){
			case 10:
			case 9:
				System.out.println("A级");
				break;
			case 8:
				System.out.println("B级");
				break;
			case 7:
				System.out.println("C级");
				break;
			case 6:
				System.out.println("D级");
				break;
			default:
				System.out.println("E级");
				break;
		}
		
	}
	
}

7.求 2/1+3/2+5/3+8/5+13/8…..前20项之和（结果的值是一个浮点类型）

/*
7、求 2/1+3/2+5/3+8/5+13/8.....前20项之和（结果的值是一个浮点类型）
*/

public class Homework7{
	
	public static void main(String[] args){
		double result = 0.0;
		double fenZi = 2;
		double fenMu = 1;
		
		for(int i=1;i<=20;i++){
			result += fenZi/fenMu;
			double temp = fenZi+fenMu;
			fenMu = fenZi;
			fenZi = temp;
		}
		System.out.println(result);
		
	}
	
}

8.打印实心等边三角形（底边*号的数量是斜边的2n-1倍）

/*
8、打印实心等边三角形（底边*号的数量是斜边的2n-1倍）
*/
import java.util.Scanner;

public class Homework8{
	
	public static void main(String[] args){
		
		Scanner scanner = new Scanner(System.in);
		System.out.print("请输入打印星号的高度: ");
		int height = scanner.nextInt();
		
		for(int i=1;i<=height;i++){
			//打印空格
			for(int j=0;j<height-i;j++){
				System.out.print(" ");
			}
			//打印星号
			for(int j=0;j<2*i-1;j++){
				System.out.print("*");
			}
			//打印换行
			System.out.println();
		}
	
	}
	
}

9.打印实心和空心菱形

/*
9、打印实心和空心菱形
	*
   * *
  *   *
 *     *
*       *
 *     *
  *   *
   * *
    *
*/
import java.util.Scanner;

public class Homework9{
	
	public static void main(String[] args){
		
		Scanner scanner = new Scanner(System.in);
		System.out.print("请输入打印星号的上半部分高度: ");
		int height = scanner.nextInt();
		
		//实心菱形
		for(int i=1;i<=height;i++){
			//打印空格
			for(int j=0;j<height-i;j++){
				System.out.print(" ");
			}
			//打印星号
			for(int j=0;j<2*i-1;j++){
				System.out.print("*");
			}
			System.out.print("\r\n");
		}
		
		//反过来打印下半部分*
		for(int i=height-1;i>0;i--){
			//打印空格
			for(int j=0;j<height-i;j++){
				System.out.print(" ");
			}
			//打印星号
			for(int j=0;j<2*i-1;j++){
				System.out.print("*");
			}
			System.out.print("\r\n");
		}
		
		
		System.out.println("---------------------");
		//空心菱形
		for(int i=1;i<=height;i++){
			//打印空格
			for(int j=1;j<=height-i;j++){
				System.out.print(" ");
			}
			//打印原本实心的星号
			for(int j=1;j<=2*i-1;j++){
				//仅在一行开始和末尾打印*
				if(j==1 || j==2*i-1){
					System.out.print("*");
				}else{
					System.out.print(" ");
				}				
			}
			System.out.print("\r\n");
		}
		
		//反过来打印下半部分*
		for(int i=height-1;i>0;i--){
			//打印空格
			for(int j=1;j<=height-i;j++){
				System.out.print(" ");
			}
			//打印原本实心的星号
			for(int j=1;j<=2*i-1;j++){
				if(j==1 || j==2*i-1){
					System.out.print("*");
				}else{
					System.out.print(" ");
				}			
			}
			System.out.print("\r\n");
		}
		
	}
	
}
